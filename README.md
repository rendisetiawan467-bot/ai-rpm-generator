model Account {
  id                String  @id @default(cuid())
  userId            String  @map("user_id")
  type              String
  provider          String
  providerAccountId String  @map("provider_account_id")
  refresh_token     String? @db.Text
  access_token      String? @db.Text
  expires_at        Int?
  token_type        String?
  scope             String?
  id_token          String? @db.Text
  session_state     String?

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, providerAccountId])
  @@map("accounts")
}

model Session {
  id           String   @id @default(cuid())
  sessionToken String   @unique @map("session_token")
  userId       String   @map("user_id")
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("sessions")
}

model User {
  id             String    @id @default(cuid())
  name           String?
  email          String    @unique
  emailVerified  DateTime? @map("email_verified")
  image          String?
  role           UserRole  @default(TEACHER)
  institution    String?
  nip            String?
  subjects       String[]
  
  accounts      Account[]
  sessions      Session[]
  rpmHistory    RPMHistory[]
  chatSessions  ChatSession[]
  
  createdAt DateTime @default(now()) @map("created_at")
  updatedAt DateTime @updatedAt @map("updated_at")

  @@map("users")
}

model VerificationToken {
  identifier String
  token      String   @unique
  expires    DateTime

  @@unique([identifier, token])
  @@map("verificationtokens")
}

enum UserRole {
  TEACHER
  SCHOOL_ADMIN
  SUPER_ADMIN
}

model RPMHistory {
  id              String   @id @default(cuid())
  userId          String   @map("user_id")
  user            User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  educationLevel  EducationLevel
  grade           String
  subject         String
  topic           String
  duration        Int
  cpElement       String   @map("cp_element")
  cpCode          String   @map("cp_code")
  cpDescription   String   @map("cp_description") @db.Text
  
  content         Json
  
  status          RPMStatus @default(DRAFT)
  shareToken      String?   @unique @map("share_token")
  shareExpiry     DateTime? @map("share_expiry")
  
  teacherSignature    String? @map("teacher_signature")
  principalSignature  String? @map("principal_signature")
  signedAt            DateTime? @map("signed_at")
  
  createdAt       DateTime  @default(now()) @map("created_at")
  updatedAt       DateTime  @updatedAt @map("updated_at")

  @@index([userId])
  @@index([shareToken])
  @@map("rpm_history")
}

enum EducationLevel {
  PAUD
  SD
  SMP
  SMA
  SMK
}

enum RPMStatus {
  DRAFT
  COMPLETED
  SHARED
  ARCHIVED
}

model ChatSession {

  messages    ChatMessage[]

model ChatMessage {
